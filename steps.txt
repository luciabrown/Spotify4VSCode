Dev build: docker build -t luciafbrown/project-full-dev:latest -f docker/Dockerfile.dev .
Dev run: docker run -it -p 12345:12345 project-full-dev
docker run -it --name dev-container -p 12345:12345 -v C:\Users\lucia\Downloads\ProjectAug25\frontend\src:/extension project-full-dev

Docker push: docker push luciafbrown/project-full:latest

Use baked-in .env (CI secrets):
docker run --name project-full -p 12345:12345 luciafbrown/project-full:latest

docker run --name project-full -p 12345:12345 --env-file ./backend/.env luciafbrown/project-full:latest
docker start -ai project-full
